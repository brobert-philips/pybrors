114                        axis=0, join="outer"

117                # Add new entry to keywords

119                    mesh = {"PMID":pmid["PMID"], "MH":self._clean_text(line)}
120                    mesh = pandas.DataFrame(data=mesh, index=[0])
121                    self.keywords = pandas.concat(
122                        [self.keywords, mesh], ignore_index=True,
123                        axis=0, join="outer"
124                    )

126                # Collect abstract

139    def _clean_text(self, text: str, keep_space: bool = True) -> str:
140        """
141        The `_clean_text` function is formatting any string.
142
143        This method replaces several groups of words by acronyms and
144        can substitute spaces by `_`. The returned string is lower
145        case.
146
147        Args:
148            text (str): Indicate the text to be cleaned.
149            keep_space (bool): Keep the spaces in the text or not.
150                Default value is True
151
152        Returns:
153            The text in lower cases, substitutes groups of words by
154            acronyms and replaces spaces by underscores
155        """
156
157        # Control text input
158        if not isinstance(text, str) or text is None:
159            err_msg = "text input value should be a string and cannot be empty."
160            raise ValueError(err_msg)
161
162        # Switch to lower cases and clean up text
163        text = text.lower().strip()
164
165        # Replace space by _
166        if keep_space:
167            return text.strip()
168        else:
169            return text.replace(" ", "_").strip()
